```json
{

    // Hover to view descriptions of existing attributes.

        {
            "name": "Python: Module",
            "type": "python",
            "request": "launch",
            "module": "${command:extension.commandvariable.file.relativeFileDotsNoExtension}",
            "justMyCode": true
        }
    ]
}
```
